488 {
489  if(m_root != nullptr) //check tree empty
490  {
491  node* delPtr = m_root;
492  T smallestInRightSubtree;
493 
494  //case 1: 0 children
495  if(m_root->left == nullptr && m_root->right == nullptr)
496  {
497  m_root = nullptr;
498  delete delPtr;
499  }
500  //case 2: 1 child
501  else if(m_root->left == nullptr && m_root->right != nullptr) //only has right child
502  {
503  m_root = m_root->right; //right child now root
504  delPtr->right = nullptr;
505  delete delPtr; //delete old root
506  }
507  else if(m_root->left != nullptr && m_root->right == nullptr) //only has left child
508  {
509  m_root = m_root->left; //left child now root
510  delPtr->left = nullptr;
511  delete delPtr; //delete old root
512  }
513  //case 3: 2 children
514  else
515  {
516  //making the smallest node in the right subtree the new root preseves the tree properties
517  smallestInRightSubtree = findSmallestPrivate(m_root->right); //store the smallest in right subtree
518  deleteNodePrivate(smallestInRightSubtree, m_root); //delete the old smallest in right subtree
519  m_root->key = smallestInRightSubtree; //make smallest in old subtree the new root
520  }

522  }
